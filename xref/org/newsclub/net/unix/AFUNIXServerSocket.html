18  package org.newsclub.net.unix;
19  
20  import java.io.IOException;
21  import java.net.ServerSocket;
22  import java.net.Socket;
23  import java.net.SocketAddress;
24  import java.net.SocketException;
25  
26  /**
27   * The server part of an AF_UNIX domain socket.
28   * 
29   * @author Christian Kohlschütter
30   */
31  public class AFUNIXServerSocket extends ServerSocket {
32    private final AFUNIXSocketImpl implementation;
33    private AFUNIXSocketAddress boundEndpoint = null;
34  
35    private final Thread shutdownThread = new Thread() {
36      @Override
37      public void run() {
38        try {
39          if (boundEndpoint != null) {
40            NativeUnixSocket.unlink(boundEndpoint.getSocketFile());
41          }
42        } catch (IOException e) {
43          // ignore
44        }
45      }
46    };
47  
48    /**
49     * Constructs a new, unconnected instance.
50     * 
51     * @throws IOException if the operation fails.
52     */
53    protected AFUNIXServerSocket() throws IOException {
54      super();
55      this.implementation = new AFUNIXSocketImpl();
56      NativeUnixSocket.initServerImpl(this, implementation);
57  
58      Runtime.getRuntime().addShutdownHook(shutdownThread);
59      NativeUnixSocket.setCreatedServer(this);
60    }
61  
62    /**
63     * Returns a new, unbound AF_UNIX {@link ServerSocket}.
64     * 
65     * @return The new, unbound {@link AFUNIXServerSocket}.
66     * @throws IOException if the operation fails.
67     */
68    public static AFUNIXServerSocket newInstance() throws IOException {
69      AFUNIXServerSockettml#AFUNIXServerSocket">AFUNIXServerSocket instance = new AFUNIXServerSocket();
70      return instance;
71    }
72  
73    /**
74     * Returns a new AF_UNIX {@link ServerSocket} that is bound to the given
75     * {@link AFUNIXSocketAddress}.
76     * 
77     * @param addr The socket file to bind to.
78     * @return The new, unbound {@link AFUNIXServerSocket}.
79     * @throws IOException if the operation fails.
80     */
81    public static AFUNIXServerSocket bindOn(final AFUNIXSocketAddress addr) throws IOException {
82      AFUNIXServerSocket socket = newInstance();
83      socket.bind(addr);
84      return socket;
85    }
86  
87    @Override
88    public void bind(SocketAddress endpoint, int backlog) throws IOException {
89      if (isClosed()) {
90        throw new SocketException("Socket is closed");
91      }
92      if (isBound()) {
93        throw new SocketException("Already bound");
94      }
95      if (!(endpoint instanceof AFUNIXSocketAddress)) {
96        throw new IOException("Can only bind to endpoints of type " + AFUNIXSocketAddress.class
97            .getName());
98      }
99      implementation.bind(backlog, endpoint);
100     boundEndpoint = (AFUNIXSocketAddress) endpoint;
101   }
102 
103   @Override
104   public boolean isBound() {
105     return boundEndpoint != null;
106   }
107 
108   @Override
109   public Socket accept() throws IOException {
110     if (isClosed()) {
111       throw new SocketException("Socket is closed");
112     }
113     AFUNIXSocket as = AFUNIXSocket.newInstance();
114     implementation.accept(as.impl);
115     as.addr = boundEndpoint;
116     NativeUnixSocket.setConnected(as);
117     return as;
118   }
119 
120   @Override
121   public String toString() {
122     if (!isBound()) {
123       return "AFUNIXServerSocket[unbound]";
124     }
125     return "AFUNIXServerSocket[" + boundEndpoint.getSocketFile() + "]";
126   }
127 
128   @Override
129   public void close() throws IOException {
130     if (isClosed()) {
131       return;
132     }
133 
134     super.close();
135     implementation.close();
136     if (boundEndpoint != null) {
137       NativeUnixSocket.unlink(boundEndpoint.getSocketFile());
138     }
139     try {
140       Runtime.getRuntime().removeShutdownHook(shutdownThread);
141     } catch (IllegalStateException e) {
142       // ignore
143     }
144   }
145 
146   /**
147    * Checks whether everything is setup to support AF_UNIX sockets.
148    * 
149    * @return {@code true} if supported.
150    */
151   public static boolean isSupported() {
152     return NativeUnixSocket.isLoaded();
153   }
154 }
